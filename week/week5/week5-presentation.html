// SPDX-License-Identifier: MIT
pragma solidity ^0.8.26;

contract HelloWorld {
    string public greet = "Hello World";
}

가장 기본적인 스마트 컨트랙트

// 확장: private 변수 + getter
contract HelloWorld {
    string private _greeting;

    function greet()
        public view
        returns (string memory)
    {
        return _greeting;
    }

    function setGreeting(
        string memory newGreeting
    ) public {
        _greeting = newGreeting;
    }
}

CheckNumber.sol

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.26;

contract CheckNumber {
    int public number = 100;

    function check(int x) public pure returns (string memory) {
        if (x > 0) return "The number is positive";
        else if (x < 0) revert("The number is negative");
        return "The number is zero";
    }
}

왜 pure?

  • 상태 변수를 읽지도 쓰지도 않음
  • 파라미터 x만으로 결과 결정
  • = 순수 함수 (Pure Function)

실행 결과

check(3) → "positive"

check(0) → "zero"

check(-5) → revert!

에러 핸들링: require / assert / revert

키워드 용도 가스 환불 사용 예시
require 입력값 / 조건 검증 O (환불) require(msg.value > 0)
assert 내부 불변식 검증 O (0.8+ 환불) assert(balance >= 0)
revert 조건부 되돌리기 O (환불) if (...) revert("err") 
// require: 조건이 false면 revert
require(msg.value > 0, "Must send ETH");

// assert: 절대 발생하면 안 되는 상황
assert(totalSupply >= 0);

// revert: if문과 함께 사용
if (balance < amount) revert("Insufficient");

Counter.sol & constructor

// SPDX-License-Identifier: MIT
pragma solidity 0.8.26;

contract Counter {
    uint public counter;

    constructor() {
        counter = 0;
    }

    function count() public {
        counter = counter + 1;
    }

    function get() public view
        returns (uint)
    {
        return counter;
    }
}

constructor 특징

  • 배포 시 딱 1회만 실행
  • 초기값 설정에 사용
  • 파라미터를 받을 수 있음
  • 배포 후에는 호출 불가

uint public counter; 기본값이 0이므로 constructor 없이도 동일!

파라미터 활용

constructor(uint _init) {
    counter = _init;
}

Faucet이란? & 핵심 상태변수

수도꼭지(Faucet) = ETH를 무료로 나눠주는 컨트랙트 (테스트넷용)

address payable public owner;

Faucet - withdraw()

function withdraw() public {
    // 1. 시간 검증 (24시간 경과 여부)
    require(
        block.timestamp >= lastWithdrawalTime[msg.sender] + LOCK_TIME,
        "Wait 24 hours"
    );
    // 2. 잔액 검증
    require(
        address(this).balance >= WITHDRAWAL_AMOUNT,
        "Not enough funds"
    );
    // 3. 출금 시간 기록
    lastWithdrawalTime[msg.sender] = block.timestamp;
    // 4. ETH 전송
    payable(msg.sender).transfer(WITHDRAWAL_AMOUNT);
    // 5. 이벤트 발행
    emit Withdrawal(msg.sender, WITHDRAWAL_AMOUNT);
}

block.timestamp = 현재 블록의 UNIX 타임스탬프 (초)

deposit / withdrawAll / receive

// ETH 입금 함수
function deposit() public payable {
    emit Deposit(msg.sender, msg.value);
}

// 전액 회수 (owner 전용)
function withdrawAll() public onlyOwner {
    uint256 amount = address(this).balance;
    owner.transfer(amount);
    emit Withdrawal(owner, amount);
}

// 직접 ETH 전송 시 자동 호출
receive() external payable {
    emit Deposit(msg.sender, msg.value);
}

payable: ETH를 받을 수 있는 함수

receive(): 데이터 없이 ETH만 보낼 때

Faucet 보안 & 사용 흐름

1. 접근 제어

onlyOwner modifier → withdrawAll은 owner만

2. 레이트 리밋

LOCK_TIME = 1 days → 24시간에 1회

3. 잔액 검증

address(this).balance → 부족 시 거부

4. 불변 파라미터

constant → 출금량/잠금시간 변경 불가

사용 흐름

1. 배포 → constructor() → owner 설정
↓
2. 입금 → deposit() / receive() → ETH 충전
↓
3. 출금 → withdraw() → 0.1 ETH 수령
↓
4. 회수 → withdrawAll() → owner 전액 회수
