🔢 Nonce

트랜잭션 일련번호. 계정마다 0부터 시작하여 TX 전송 시 1씩 증가합니다. 동일 nonce를 재사용할 수 없어 리플레이 공격을 방지합니다.

비유: 은행 거래 일련번호

⛽ Gas

EVM 연산 비용 단위. 복잡할수록 Gas가 많이 소모됩니다. 스팸 방지 + 검증자 보상 두 가지 역할을 합니다.

비유: 자동차 연료

✍ Signature

ECDSA 디지털 서명 (r, s, v). 개인키로 서명하고 공개키로 검증합니다. 인증 + 무결성 + 부인방지를 보장합니다.

비유: 위조 불가 도장

🌉 Bridge

L1↔L2 간 자산 이동. L1에서 잠그고 L2에서 발행하거나, L2에서 소각하고 L1에서 해제합니다.

비유: 국제 송금 (환전소)

브릿지 개념

Deposit (L1→L2): L1에서 자산을 잠그고(Lock), L2에서 동일 자산을 발행(Mint)합니다.

Withdraw (L2→L1): L2에서 자산을 소각(Burn)하고, L1에서 잠긴 자산을 해제(Unlock)합니다.

⚠ L2→L1 출금은 7일 챌린지 기간이 필요합니다. 이 기간 동안 사기 증명(Fraud Proof)이 제출될 수 있습니다.

각 스크립트의 코드 템플릿입니다. ethers.js v6 + dotenv 패턴을 사용합니다. TODO 부분을 완성하세요.

01-check-balance.js — 잔액 + Nonce 조회 
// 01-check-balance.js
const { ethers } = require('ethers');
require('dotenv').config();

const provider = new ethers.JsonRpcProvider(process.env.RPC_URL);
const address = 'YOUR_WALLET_ADDRESS'; // TODO: 본인 지갑 주소 입력

async function main() {
  const balance = await provider.getBalance(address);
  console.log('Balance:', ethers.formatEther(balance), 'ETH');
  const nonce = await provider.getTransactionCount(address);
  console.log('Nonce:', nonce);
}
main().catch(console.error);
02-send-eth.js — ETH 송금 
// 02-send-eth.js
const { ethers } = require('ethers');
require('dotenv').config();

const provider = new ethers.JsonRpcProvider(process.env.RPC_URL);
const wallet = new ethers.Wallet(process.env.PRIVATE_KEY, provider);

async function main() {
  const to = 'RECIPIENT_ADDRESS'; // TODO: 받는 주소 입력
  const amount = '0.001';         // TODO: 송금할 ETH 수량

  console.log('송금 전 nonce:', await provider.getTransactionCount(wallet.address));

  const tx = await wallet.sendTransaction({
    to: to,
    value: ethers.parseEther(amount),
  });
  console.log('TX Hash:', tx.hash);

  const receipt = await tx.wait();
  console.log('Status:', receipt.status === 1 ? '성공' : '실패');
  console.log('Gas Used:', receipt.gasUsed.toString());
}
main().catch(console.error);
03-track-tx.js — TX 추적 (Nonce, Gas, Signature r/s/v) 
// 03-track-tx.js
const { ethers } = require('ethers');
require('dotenv').config();

const provider = new ethers.JsonRpcProvider(process.env.RPC_URL);

async function main() {
  const txHash = 'YOUR_TX_HASH'; // TODO: 조회할 TX 해시 입력

  const tx = await provider.getTransaction(txHash);
  const receipt = await provider.getTransactionReceipt(txHash);

  console.log('=== 기본 정보 ===');
  console.log('From:', tx.from);
  console.log('To:', tx.to);
  console.log('Value:', ethers.formatEther(tx.value), 'ETH');

  console.log('\n=== Nonce ===');
  console.log('Nonce:', tx.nonce);

  console.log('\n=== Gas ===');
  console.log('Gas Limit:', tx.gasLimit.toString());
  if (receipt) {
    console.log('Gas Used:', receipt.gasUsed.toString());
    const fee = receipt.gasUsed * receipt.gasPrice;
    console.log('수수료:', ethers.formatEther(fee), 'ETH');
  }

  console.log('\n=== Signature (r, s, v) ===');
  console.log('r:', tx.signature.r);
  console.log('s:', tx.signature.s);
  console.log('v:', tx.signature.v);
}
main().catch(console.error);
04-verify-sig.js — 서명 검증 (ecrecover) 
// 04-verify-sig.js
const { ethers } = require('ethers');
require('dotenv').config();

const provider = new ethers.JsonRpcProvider(process.env.RPC_URL);

async function main() {
  const txHash = 'YOUR_TX_HASH'; // TODO: 검증할 TX 해시 입력

  const tx = await provider.getTransaction(txHash);

  // 서명에서 주소 복원 (ecrecover)
  const unsignedTx = {
    to: tx.to,
    value: tx.value,
    nonce: tx.nonce,
    gasLimit: tx.gasLimit,
    gasPrice: tx.gasPrice,
    data: tx.data,
    chainId: tx.chainId,
    type: tx.type,
    maxFeePerGas: tx.maxFeePerGas,
    maxPriorityFeePerGas: tx.maxPriorityFeePerGas,
  };
  const serialized = ethers.Transaction.from(unsignedTx).unsignedSerialized;
  const digest = ethers.keccak256(serialized);

  const recoveredAddress = ethers.recoverAddress(digest, tx.signature);

  console.log('TX From:', tx.from);
  console.log('Recovered:', recoveredAddress);
  console.log('일치 여부:', tx.from.toLowerCase() === recoveredAddress.toLowerCase());
}
main().catch(console.error);
05-bridge-tx.js — Bridge TX 분석 
// 05-bridge-tx.js
const { ethers } = require('ethers');
require('dotenv').config();

// L1과 L2 프로바이더
const l1Provider = new ethers.JsonRpcProvider(process.env.L1_RPC_URL || 'https://ethereum-sepolia-rpc.publicnode.com');
const l2Provider = new ethers.JsonRpcProvider(process.env.RPC_URL);

async function main() {
  const txHash = 'YOUR_BRIDGE_TX_HASH'; // TODO: 브릿지 TX 해시 입력

  // TODO: 해당 네트워크의 프로바이더로 조회
  const tx = await l2Provider.getTransaction(txHash);
  const receipt = await l2Provider.getTransactionReceipt(txHash);

  console.log('=== Bridge TX 분석 ===');
  console.log('From:', tx.from);
  console.log('To:', tx.to);
  console.log('Value:', ethers.formatEther(tx.value), 'ETH');
  console.log('Nonce:', tx.nonce);

  if (receipt) {
    console.log('Status:', receipt.status === 1 ? '성공' : '실패');
    console.log('Gas Used:', receipt.gasUsed.toString());
  }

  // 컨트랙트 호출 데이터 확인
  if (tx.data && tx.data !== '0x') {
    console.log('\n=== 컨트랙트 상호작용 ===');
    console.log('Function Selector:', tx.data.slice(0, 10));
    console.log('Data Length:', (tx.data.length - 2) / 2, 'bytes');
  } else {
    console.log('\n일반 ETH 전송 (컨트랙트 호출 없음)');
  }

  // 서명 정보
  console.log('\n=== Signature ===');
  console.log('r:', tx.signature.r);
  console.log('s:', tx.signature.s);
  console.log('v:', tx.signature.v);
}
main().catch(console.error);
